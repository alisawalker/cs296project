166  void CreateCircle()
167  {
168  float whycoa = 1.5, mla = 1.41421, dla = 2.82843, cla = 8.48528, rodWidtha = 1, frontHeighta = 8, secondHeighta = 4;
169  float radius = 0.2/2;
170  b2CircleShape shape;
171  shape.m_p.Set(0,0);
172  b2Vec2 p;
173  // std::cout << excoa << whycoa << rodWidtha << frontHeighta << secondHeighta << std::endl;
174  if((*pumpingRod).GetLinearVelocity().x - (*frontl).GetLinearVelocity().x > 0){
175  p.Set((*frontl).GetWorldCenter().x - 0.5 + mla + dla/2, whycoa + rodWidtha + frontHeighta + rodWidtha + secondHeighta + 1);
176  }
177  else p.Set((*frontl).GetWorldCenter().x - 0.5 + mla + dla + cla + dla/2, whycoa + rodWidtha + frontHeighta + rodWidtha + secondHeighta + 1);
178  shape.m_radius = radius;
179  b2FixtureDef fd;
180  fd.shape = &shape;
181  fd.restitution = 1;
182  fd.density = 80;
183 // fd.friction = 0.0f;
184  fd.filter.maskBits = 0x0004;
185  fd.filter.categoryBits = 0x0002;
186 
187 // b2Vec2 p(rand()%10, rand()%10);
188  b2BodyDef bd;
189  bd.bullet = true;
190  for(int i = 0; i < 10; i++){
191  bd.linearVelocity.Set(rand()%10, rand()%10);
192  bd.type = b2_dynamicBody;
193  bd.position = p;

196  (*body).SetGravityScale(0.1);
